    public int getTotalBoxes() {
        int countOfBoxes = 0;

        for (int i = 0; i < orders.size(); i++) {
            countOfBoxes += orders.get(i).getNumBoxes();
        }

        return countOfBoxes;
    }

    public int removeVariety(String cookieVar) {
        int numRemoved = 0;

        //Iterate through all orders and localize findings
        for (int i = 0; i < orders.size(); i++) {
            String variety = orders.get(i).getVariety();
            //If variety equals the wanted variety, remove and increment count
            if (cookieVar.equals(variety)) {
                orders.remove(i);
                numRemoved++;
            }
        }
        
        return numRemoved;
    }
